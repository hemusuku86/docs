---
title: "Verify API"
description: "Verify your tokens."
---

## Endpoint

`POST` https://hamutan86.pythonanywhere.com/nonstress/api/verify

## Request

<Columns cols={2}>
  <Card>
    <ParamField path="token" type="string">
      the nonstress token you received from client-side.
    </ParamField>
  </Card>
  <Card>
    <Tabs>
      <Tab title="Verified">
      ```json JSON
      {
        "token": "a5Eg2Jqjh+F177T3wmnSbarx72Qxbkz8YjZH..."
      }
      ```
      </Tab>
    </Tabs>
  </Card>
</Columns>

## Response

<Columns cols={2}>
  <Card>
    <ParamField path="error" type="string / null">
      will be the error text when an error occurred in verification
    </ParamField>
    <ParamField path="security[pass]" type="boolean">
      Did the user passed our verification or not.
    </ParamField>
    <ParamField path="security[score]" type="float">
      the suspicion score of user. higher score = more human, lower score = more bot. (0.0-1.0)
    </ParamField>
    <ParamField path="security[url]" type="string">
      the url where the token generated. you can block user if url didn't match with your expect.
    </ParamField>
    <ParamField path="user[device_fingerprint]" type="string">
      user's device fingerprint ID.
    </ParamField>
    <ParamField path="user[device_values]" type="JSON">
      Pro Plan Only; user's device fingerprint values.
    </ParamField>
    <ParamField path="user[ip]" type="string">
      user's IP address.
    </ParamField>
    <ParamField path="user[user-agent]" type="string">
      user's User-Agent header.
    </ParamField>
  </Card>
  <Card>
    <Tabs>
      <Tab title="Verified">
      ```json
      {
          "error": null,
          "security": {
              "pass": true,
              "score": 1.0,
              "url": "https://hamutan86.pythonanywhere.com/nonstress/demo"
          },
          "user": {
              "device_fingerprint": "677d41785ab...",
              // "device_values": {"lang": "en"...}, (Pro Plan Only)
              "ip": "1.2.3.4",
              "user-agent": "Mozilla/5.0 (Windows NT 10.0; ..."
          }
      }
      ```
      </Tab>
      <Tab title="Error">
      ```json
      {
          "error": "invalid token",
          "security": {
              "pass": false
          }
      }
      ```
      </Tab>
      <Tab title="Bot Detected">
      ```json
      {
          "error": null,
          "security": {
              "pass": false,
              "score": 0.0,
              "url": "https://hamutan86.pythonanywhere.com/nonstress/demo"
          },
          "user": {
              "device_fingerprint": "b9d721c204a...",
              // "device_values": {"lang": "en"...}, (Pro Plan Only)
              "ip": "4.3.2.1",
              "user-agent": "Mozilla/5.0 (Windows NT 10.0; ..."
          }
      }
      ```
      </Tab>
    </Tabs>
  </Card>
</Columns>
